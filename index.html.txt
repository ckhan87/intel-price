import React, { useState, useEffect } from 'react';
import { 
  Newspaper, 
  BrainCircuit, 
  Settings, 
  ChevronRight, 
  ExternalLink, 
  Loader2, 
  AlertCircle,
  Key,
  TrendingUp,
  Sparkles,
  Cpu,
  Swords
} from 'lucide-react';

// Realistic mock data for Intel News to ensure the app works out-of-the-box
// In a production app, this would be replaced with a fetch to a real news API
const MOCK_NEWS = [
  {
    id: 1,
    title: "Intel Unveils Gaudi 3 AI Accelerator to Challenge Nvidia",
    date: "2026-03-12",
    summary: "Intel has officially launched its highly anticipated Gaudi 3 AI accelerator, promising significant performance-per-dollar advantages over competitors. The new chip targets enterprise AI workloads and large language model inference.",
    url: "#"
  },
  {
    id: 2,
    title: "Intel Foundry Achieves Key Milestone on 14A Process Node",
    date: "2026-03-10",
    summary: "Marking a major step in its IDM 2.0 strategy, Intel Foundry announced early test chip success on its advanced 14A process node, utilizing High-NA EUV lithography ahead of its competitors.",
    url: "#"
  },
  {
    id: 3,
    title: "Intel and Microsoft Expand Partnership on AI PCs",
    date: "2026-03-08",
    summary: "A new joint initiative aims to deeply integrate Windows Copilot features directly into Intel's next-generation Core Ultra processors, bringing advanced local AI capabilities to millions of enterprise users.",
    url: "#"
  },
  {
    id: 4,
    title: "Quarterly Earnings: Intel Shows Recovery in Client Computing",
    date: "2026-03-01",
    summary: "Intel's Q1 financial results indicate a strong rebound in the PC market, driven by the refresh cycle and the introduction of AI-enabled PCs. Data center revenues remain highly competitive.",
    url: "#"
  },
  {

    title: "Intel Announces Expansion of Packaging Facilities in Advanced Hub",

    summary: "To meet growing demand for advanced packaging technologies like Foveros, Intel is committing an additional $2 billion to expand its packaging and testing facilities globally.",
    url: "#"
  }
];

export default function App() {
  const [news, setNews] = useState([]);
  const [selectedArticle, setSelectedArticle] = useState(null);
  const [intcPrice, setIntcPrice] = useState(null);
  const [isLoadingPrice, setIsLoadingPrice] = useState(false);
  
  // Quick Insights AI State
  const [quickInsight, setQuickInsight] = useState({ type: null, text: '', loading: false, error: '' });

  // DeepSeek AI State - Initialized with your provided key
  const [apiKey, setApiKey] = useState('sk-607570af647e4ce3baabf5f3da383adf');
  const [showSettings, setShowSettings] = useState(false);
  const [aiResponse, setAiResponse] = useState('');
  const [isLoadingAi, setIsLoadingAi] = useState(false);
  const [aiError, setAiError] = useState('');

  const fetchPrice = async () => {
    setIsLoadingPrice(true);
    try {
      // Attempt 1: Using CodeTabs proxy (more reliable than allorigins)
      const res = await fetch('https://api.codetabs.com/v1/proxy?quest=https://query2.finance.yahoo.com/v8/finance/chart/INTC');
      if (!res.ok) throw new Error('Primary proxy failed');
      const data = await res.json();
      const price = data.chart.result[0].meta.regularMarketPrice;
      setIntcPrice(price.toFixed(2));
    } catch (error) {
      try {
        // Attempt 2: Using corsproxy.io as a fallback
        const targetUrl = encodeURIComponent('https://query2.finance.yahoo.com/v8/finance/chart/INTC');
        const fallbackRes = await fetch(`https://corsproxy.io/?${targetUrl}`);
        if (!fallbackRes.ok) throw new Error('Fallback proxy failed');
        const data = await fallbackRes.json();
        const price = data.chart.result[0].meta.regularMarketPrice;
        setIntcPrice(price.toFixed(2));
      } catch (fallbackError) {
        // Suppress massive console errors and use a clean log + fallback
        console.log("Live price fetch restricted by network. Using static fallback.");
        setIntcPrice("45.77");
      }
    } finally {
      setIsLoadingPrice(false);
    }
  };

  useEffect(() => {
    // Simulate fetching news
    setNews(MOCK_NEWS);
    setSelectedArticle(MOCK_NEWS[0]);
    
    // Fetch live INTC price
    fetchPrice();
    
    // Try to load API key from local storage for convenience, but default to the hardcoded one
    const savedKey = localStorage.getItem('deepseek_api_key');
    if (savedKey) setApiKey(savedKey);
  }, []);

  const handleSaveKey = (e) => {
    e.preventDefault();
    localStorage.setItem('deepseek_api_key', apiKey);
    setShowSettings(false);
  };

  const getAiSuggestion = async () => {
    if (!apiKey) {
      setShowSettings(true);
      setAiError("Please configure your DeepSeek API key first.");
      return;
    }

    if (!selectedArticle) return;

    setIsLoadingAi(true);
    setAiResponse('');
    setAiError('');

    try {
      const response = await fetch('https://api.deepseek.com/chat/completions', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${apiKey}`
        },
        body: JSON.stringify({
          model: 'deepseek-chat',
          messages: [
            {
              role: 'system',
              content: 'You are an expert options trading analyst. Analyze the provided news article summary and the current Intel (INTC) stock price. Provide a specific technical and strategic opinion on whether a trader should buy INTC call options right now. Provide 3 highly specific, actionable insights, potentially suggesting strike prices, expirations, or technical setups based on this catalyst. Format your response with clear bullet points. Keep it concise, professional, and forward-looking.'
            },
            {
              role: 'user',
              content: `Current INTC Stock Price: $${intcPrice || 'Unknown'}\n\nNews Title: ${selectedArticle.title}\nDate: ${selectedArticle.date}\nSummary: ${selectedArticle.summary}`
            }
          ],
          temperature: 0.7,
          max_tokens: 500
        })
      });

      const data = await response.json();

      if (!response.ok) {
        throw new Error(data.error?.message || `API Error: ${response.status}`);
      }

      setAiResponse(data.choices[0].message.content);
    } catch (err) {
      console.error("DeepSeek API Error:", err);
      setAiError(err.message || "Failed to fetch AI suggestions. Please check your API key and network connection.");
    } finally {
      setIsLoadingAi(false);
    }
  };

  const handleQuickAction = async (actionType) => {
    if (!apiKey) {
      setShowSettings(true);
      setQuickInsight({ type: null, text: '', loading: false, error: 'Please configure your DeepSeek API key first.' });
      return;
    }

    if (!selectedArticle) return;
    
    setQuickInsight({ type: actionType, text: '', loading: true, error: '' });
    
    let systemPrompt = "";
    if (actionType === 'explain') {
      systemPrompt = "You are a tech educator. Explain the complex semiconductor or AI technology mentioned in this article in simple terms (Explain Like I'm 5). Keep it to 2-3 short paragraphs.";
    } else if (actionType === 'competitors') {
      systemPrompt = "You are a semiconductor industry analyst. Briefly analyze how this Intel news impacts its main competitors (like AMD, Nvidia, TSMC, ARM). Use bullet points and keep it concise.";
    }

    try {
      const response = await fetch('https://api.deepseek.com/chat/completions', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${apiKey}`
        },
        body: JSON.stringify({
          model: 'deepseek-chat',
          messages: [
            { role: 'system', content: systemPrompt },
            { role: 'user', content: `Title: ${selectedArticle.title}\nSummary: ${selectedArticle.summary}` }
          ],
          temperature: 0.7,
          max_tokens: 500
        })
      });

      const data = await response.json();

      if (!response.ok) {
        throw new Error(data.error?.message || `API Error: ${response.status}`);
      }

      const generatedText = data.choices[0].message.content;
      setQuickInsight({ type: actionType, text: generatedText, loading: false, error: '' });
    } catch (err) {
      console.error("DeepSeek API Error:", err);
      setQuickInsight(prev => ({ ...prev, loading: false, error: "Failed to load DeepSeek insights. " + err.message }));
    }
  };

  // Select a new article and clear previous AI insights
  const handleSelectArticle = (article) => {
    setSelectedArticle(article);
    setAiResponse('');
    setAiError('');
    setQuickInsight({ type: null, text: '', loading: false, error: '' });
  };

  return (
    <div className="min-h-screen bg-slate-50 text-slate-900 font-sans selection:bg-blue-200">
      {/* Navigation Header */}
      <header className="bg-blue-900 text-white shadow-md sticky top-0 z-10">
        <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
          <div className="flex justify-between items-center h-16">
            <div className="flex items-center space-x-3">
              <div className="bg-blue-600 p-2 rounded-lg">
                <Newspaper className="w-6 h-6 text-white" />
              </div>
              <h1 className="text-xl font-bold tracking-tight">Intel<span className="text-blue-300">Pulse</span> AI</h1>
            </div>
            <div className="flex items-center space-x-4">
              <div className="flex items-center bg-blue-800/80 border border-blue-700 px-3 py-1.5 rounded-lg text-sm font-medium shadow-sm">
                <TrendingUp className="w-4 h-4 mr-2 text-emerald-400" />
                <span className="text-blue-200 mr-1">INTC:</span>
                <span className="font-bold text-white">
                  {isLoadingPrice ? 'Loading...' : `$${intcPrice}`}
                </span>
              </div>
              <button 
                onClick={() => setShowSettings(true)}
                className="flex items-center space-x-2 bg-blue-800 hover:bg-blue-700 transition-colors px-4 py-2 rounded-md text-sm font-medium"
              >
                <Settings className="w-4 h-4" />
                <span className="hidden sm:inline">API Settings</span>
              </button>
            </div>
          </div>
        </div>
      </header>

      {/* Main Content Layout */}
      <main className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-8 h-[calc(100vh-4rem)]">
        <div className="flex flex-col lg:flex-row gap-6 h-full">
          
          {/* Left Column: News Feed */}
          <div className="w-full lg:w-1/3 flex flex-col h-full">
            <h2 className="text-lg font-semibold mb-4 text-slate-700 flex items-center">
              <Newspaper className="w-5 h-5 mr-2 text-blue-600"/> Latest News
            </h2>
            <div className="flex-1 overflow-y-auto space-y-4 pr-2 custom-scrollbar">
              {news.map((article) => (
                <div 
                  key={article.id}
                  onClick={() => handleSelectArticle(article)}
                  className={`p-4 rounded-xl border transition-all cursor-pointer ${
                    selectedArticle?.id === article.id 
                      ? 'bg-blue-50 border-blue-400 shadow-sm' 
                      : 'bg-white border-slate-200 hover:border-blue-300 hover:shadow-sm'
                  }`}
                >
                  <div className="text-xs font-medium text-slate-500 mb-2">{article.date}</div>
                  <h3 className={`font-bold leading-tight mb-2 ${selectedArticle?.id === article.id ? 'text-blue-900' : 'text-slate-800'}`}>
                    {article.title}
                  </h3>
                  <p className="text-sm text-slate-600 line-clamp-2">
                    {article.summary}
                  </p>
                </div>
              ))}
            </div>
          </div>

          {/* Right Column: Article Details & AI Analyst */}
          <div className="w-full lg:w-2/3 flex flex-col h-full">
            {selectedArticle ? (
              <div className="bg-white rounded-2xl border border-slate-200 shadow-sm flex flex-col h-full overflow-hidden">
                
                {/* Article Header */}
                <div className="p-6 md:p-8 border-b border-slate-100 bg-slate-50/50">
                  <div className="inline-block px-3 py-1 rounded-full bg-slate-200 text-slate-700 text-xs font-semibold mb-4">
                    {selectedArticle.date}
                  </div>
                  <h2 className="text-2xl md:text-3xl font-extrabold text-slate-900 mb-4 leading-tight">
                    {selectedArticle.title}
                  </h2>
                  <p className="text-slate-700 text-lg leading-relaxed">
                    {selectedArticle.summary}
                  </p>
                  <a href={selectedArticle.url} className="inline-flex items-center mt-4 text-blue-600 hover:text-blue-800 font-medium">
                    Read full article <ExternalLink className="w-4 h-4 ml-1" />
                  </a>

                  {/* Quick Insights Section */}
                  <div className="mt-8 pt-6 border-t border-slate-200">
                    <h4 className="text-sm font-bold text-slate-500 uppercase tracking-wider mb-3 flex items-center">
                      <Sparkles className="w-4 h-4 mr-2 text-amber-500" />
                      DeepSeek Quick Insights
                    </h4>
                    <div className="flex flex-wrap gap-3 mb-4">
                      <button
                        onClick={() => handleQuickAction('explain')}
                        disabled={quickInsight.loading}
                        className={`flex items-center px-4 py-2 rounded-lg text-sm font-medium transition-colors border ${
                          quickInsight.type === 'explain' ? 'bg-amber-50 border-amber-300 text-amber-800' : 'bg-white border-slate-200 text-slate-700 hover:bg-slate-50 hover:border-amber-200'
                        }`}
                      >
                        {quickInsight.loading && quickInsight.type === 'explain' ? (
                          <Loader2 className="w-4 h-4 mr-2 animate-spin text-amber-600" />
                        ) : (
                          <Cpu className="w-4 h-4 mr-2 text-amber-600" />
                        )}
                        Explain the Tech ✨
                      </button>
                      
                      <button
                        onClick={() => handleQuickAction('competitors')}
                        disabled={quickInsight.loading}
                        className={`flex items-center px-4 py-2 rounded-lg text-sm font-medium transition-colors border ${
                          quickInsight.type === 'competitors' ? 'bg-rose-50 border-rose-300 text-rose-800' : 'bg-white border-slate-200 text-slate-700 hover:bg-slate-50 hover:border-rose-200'
                        }`}
                      >
                        {quickInsight.loading && quickInsight.type === 'competitors' ? (
                          <Loader2 className="w-4 h-4 mr-2 animate-spin text-rose-600" />
                        ) : (
                          <Swords className="w-4 h-4 mr-2 text-rose-600" />
                        )}
                        Competitor Impact ✨
                      </button>
                    </div>

                    {/* AI Result Display */}
                    {quickInsight.error && (
                      <div className="text-sm text-red-600 bg-red-50 p-3 rounded-lg border border-red-100 mb-4">
                        {quickInsight.error}
                      </div>
                    )}
                    
                    {quickInsight.text && !quickInsight.loading && (
                      <div className="bg-white border border-slate-200 p-5 rounded-xl text-slate-700 text-sm shadow-sm">
                        {quickInsight.text.split('\n').map((line, i) => {
                          if (line.trim() === '') return <br key={i} />;
                          const parts = line.split(/(\*\*.*?\*\*)/g);
                          return (
                            <p key={i} className="mb-2">
                              {parts.map((part, j) => {
                                if (part.startsWith('**') && part.endsWith('**')) {
                                  return <strong key={j} className="text-slate-900">{part.slice(2, -2)}</strong>;
                                }
                                return part;
                              })}
                            </p>
                          );
                        })}
                      </div>
                    )}
                  </div>
                </div>

                {/* AI Analyst Section */}
                <div className="p-6 md:p-8 flex-1 flex flex-col bg-slate-50/50 overflow-y-auto border-t border-slate-200">
                  <div className="flex items-center justify-between mb-6">
                    <h3 className="text-xl font-bold text-slate-800 flex items-center">
                      <BrainCircuit className="w-6 h-6 mr-2 text-purple-600" />
                      Options Analysis
                    </h3>
                    <button
                      onClick={getAiSuggestion}
                      disabled={isLoadingAi}
                      className="bg-purple-600 hover:bg-purple-700 disabled:bg-purple-400 text-white px-5 py-2.5 rounded-lg font-medium flex items-center transition-colors shadow-sm"
                    >
                      {isLoadingAi ? (
                        <Loader2 className="w-5 h-5 mr-2 animate-spin" />
                      ) : (
                        <TrendingUp className="w-5 h-5 mr-2" />
                      )}
                      {isLoadingAi ? 'Analyzing...' : 'Analyze Call Options'}
                    </button>
                  </div>

                  {/* Errors */}
                  {aiError && (
                    <div className="bg-red-50 border border-red-200 text-red-700 p-4 rounded-xl flex items-start mb-6">
                      <AlertCircle className="w-5 h-5 mr-3 mt-0.5 flex-shrink-0" />
                      <div className="text-sm">{aiError}</div>
                    </div>
                  )}

                  {/* AI Response Area */}
                  {aiResponse ? (
                    <div className="prose prose-slate max-w-none text-slate-700 bg-purple-50/30 p-6 rounded-xl border border-purple-100">
                      {/* Simple Markdown rendering for bullets and bold text */}
                      {aiResponse.split('\n').map((line, i) => {
                        if (line.trim() === '') return <br key={i} />;
                        // Handle basic markdown bold (**)
                        const parts = line.split(/(\*\*.*?\*\*)/g);
                        return (
                          <p key={i} className="mb-2">
                            {parts.map((part, j) => {
                              if (part.startsWith('**') && part.endsWith('**')) {
                                return <strong key={j} className="text-slate-900">{part.slice(2, -2)}</strong>;
                              }
                              return part;
                            })}
                          </p>
                        );
                      })}
                    </div>
                  ) : (
                    !isLoadingAi && !aiError && (
                      <div className="flex flex-col items-center justify-center text-center p-12 border-2 border-dashed border-slate-200 rounded-xl bg-slate-50 flex-1">
                        <div className="bg-purple-100 p-4 rounded-full mb-4">
                          <BrainCircuit className="w-8 h-8 text-purple-600" />
                        </div>
                        <h4 className="text-lg font-semibold text-slate-800 mb-2">Ready for Options Analysis</h4>
                        <p className="text-slate-500 max-w-sm">
                          Click the button above to generate a technical opinion and call option strategies using the latest INTC price and DeepSeek API.
                        </p>
                      </div>
                    )
                  )}
                </div>
              </div>
            ) : (
              <div className="flex items-center justify-center h-full text-slate-500">
                Select an article to view details
              </div>
            )}
          </div>
        </div>
      </main>

      {/* Settings Modal */}
      {showSettings && (
        <div className="fixed inset-0 bg-slate-900/50 backdrop-blur-sm flex items-center justify-center z-50 px-4">
          <div className="bg-white rounded-2xl shadow-xl w-full max-w-md overflow-hidden animate-in fade-in zoom-in duration-200">
            <div className="p-6 border-b border-slate-100 flex justify-between items-center">
              <h3 className="text-xl font-bold flex items-center text-slate-800">
                <Key className="w-5 h-5 mr-2 text-slate-500" />
                API Configuration
              </h3>
              <button 
                onClick={() => setShowSettings(false)}
                className="text-slate-400 hover:text-slate-600 transition-colors"
              >
                ✕
              </button>
            </div>
            <form onSubmit={handleSaveKey} className="p-6">
              <div className="mb-4">
                <label className="block text-sm font-semibold text-slate-700 mb-2">
                  DeepSeek API Key
                </label>
                <input
                  type="password"
                  value={apiKey}
                  onChange={(e) => setApiKey(e.target.value)}
                  placeholder="sk-..."
                  className="w-full px-4 py-3 rounded-lg border border-slate-300 focus:ring-2 focus:ring-blue-500 focus:border-blue-500 outline-none transition-all"
                  required
                />
                <p className="text-xs text-slate-500 mt-2">
                  Your key is stored locally in your browser's localStorage and is only sent directly to DeepSeek's API.
                </p>
              </div>
              <div className="flex justify-end space-x-3 mt-8">
                <button
                  type="button"
                  onClick={() => setShowSettings(false)}
                  className="px-4 py-2 text-slate-600 hover:bg-slate-100 rounded-lg font-medium transition-colors"
                >
                  Cancel
                </button>
                <button
                  type="submit"
                  className="px-4 py-2 bg-blue-600 hover:bg-blue-700 text-white rounded-lg font-medium transition-colors"
                >
                  Save Settings
                </button>
              </div>
            </form>
          </div>
        </div>
      )}
      
      {/* Global CSS for custom scrollbar */}
      <style dangerouslySetInnerHTML={{__html: `
        .custom-scrollbar::-webkit-scrollbar {
          width: 6px;
        }
        .custom-scrollbar::-webkit-scrollbar-track {
          background: transparent;
        }
        .custom-scrollbar::-webkit-scrollbar-thumb {
          background-color: #cbd5e1;
          border-radius: 20px;
        }
      `}} />
    </div>
  );
}
